CREATE TABLE Professores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(255),
    escola VARCHAR(255),
    data_contratacao DATE,
    turmas VARCHAR(255),
    Usuario_ID INT,
    FOREIGN KEY (Usuario_ID) REFERENCES Usuarios(ID)
);
